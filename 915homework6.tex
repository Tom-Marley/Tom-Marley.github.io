\noindent
Throughout, $R$ denotes a commutative ring with identity.

\medskip


\begin{problem} Prove that an $R$-module $E$ is injective if and only if $\Ext^1_R(R/I,E)=0$ for all ideals $I$ of $R$.
\end{problem}

\begin{problem} Suppose for some integer $n$ that $\pd_R R/I\le n$ for all ideals $I$ of $R$.  Prove that $\id_R M\le n$ for all $R$-modules $M$.  
\end{problem}

\begin{problem} Let $R$ be a Noetherian ring and $M$ an $R$-module.  Prove that $M$ is an injective $R$-module if and only if $M_p$ is an injective $R_p$-module for all prime ideals $p$ of $R$.
\end{problem}


\begin{problem} Consider the contravariant functor $(-)^*:=\Hom_{\Z}(-,\Q/\Z)$.  As $\Q/\Z$ is an injective $\Z$-module, $(-)^*$ is an exact functor from $R$-mod to $R$-mod.  (Note that every $R$-module is a $\Z-R$ bimodule, so $M^*$ is an $R$-module for every $R$-module $M$.)
\begin{enumerate}[(a)]
\item Prove that the evaluation homomorphism $\phi:M\to M^{**}$ is an injective $R$-module homomorphism for every $R$-module $M$.  (The evaluation homomorphism is defined as follows:  For $u\in M$, define $\phi(u)(f):=f(u)$ for every $f\in M^*$.)
\item Let $f:A\to B$ and $g:B\to C$ be $R$-module homomorphisms.  Prove that $A\to B\to C$ is exact if and only if $C^*\to B^*\to A^*$ is exact.
\item Prove that $M$ is a flat $R$-module if and only if $M^*$ is an injective $R$-module.
\end{enumerate}
\end{problem}


\begin{problem} Let $(R,m)$ be a local PID which is not a field. Prove that $E_R(R/m)\cong Q/R$, where
$Q$ is the field of fractions of $R$.  
\end{problem}

\begin{problem} Let $I$ be ideal of $R$, and $M$ an $R$-module.
Prove that $\Hom_R(R/I,E_R(M))\cong E_{R/I}(\Hom_R(R/I,M))$.
\end{problem}

\begin{problem} Let $(R,m)$ be a quasi-local ring.
\begin{enumerate}[(a)]
\item Prove that $R$ is indecomposable (as an $R$-module).
\item Suppose $R$ is Noetherian and an injective $R$-module.  Prove that $R\cong E_R(R/m)$.)
\end{enumerate}
\end{problem}
